  const [templates, setTemplates] = useLocalStorage<TaskTemplate[]>(
    'jobhunter-templates',
    DEFAULT_TEMPLATES
  )

  const createFromTemplate = (templateId: string) => {
    const template = templates.find(t => t.id === templateId)
    if (!template) return null

    // æ›´æ–°ä½¿ç”¨ç»Ÿè®¡

  }

  const saveAsTemplate = (taskData, name, icon) => {
    const newTemplate: TaskTemplate = {
      id: Date.now().toString(),
      name,
      icon,
      taskData,
      isDefault: false,
      usageCount: 0,
      lastUsed: new Date()
    }
    setTemplates([...templates, newTemplate])
  }

  return { templates, createFromTemplate, saveAsTemplate }
}

// hooks/useWeekTasks.ts
export function useWeekTasks(tasks: Task[]) {
  const [currentWeekStart, setCurrentWeekStart] = useState(() =>
    getWeekStart(new Date())
  )

  const weekTasks = useMemo(() => {
    const weekEnd = new Date(currentWeekStart)
    weekEnd.setDate(weekEnd.getDate() + 7)

    return tasks.filter(task => {
      const date = task.scheduledDate || task.createdAt
      return date >= currentWeekStart && date < weekEnd
    })
  }, [tasks, currentWeekStart])

  const tasksByDay = useMemo(() => {
    const grouped: Record<string, Task[]> = {}

    for (let i = 0; i < 7; i++) {
      const day = new Date(currentWeekStart)
      day.setDate(day.getDate() + i)
      const key = day.toISOString().split('T')[0]
      grouped[key] = []
    }

    weekTasks.forEach(task => {
      const date = task.scheduledDate || task.createdAt
      const key = date.toISOString().split('T')[0]
      if (grouped[key]) {
        grouped[key].push(task)
      }
    })

    return grouped
  }, [weekTasks, currentWeekStart])

  const goToNextWeek = () => {
    const next = new Date(currentWeekStart)
    next.setDate(next.getDate() + 7)
    setCurrentWeekStart(next)
  }

  const goToPrevWeek = () => {
    const prev = new Date(currentWeekStart)
    prev.setDate(prev.getDate() - 7)
    setCurrentWeekStart(prev)
  }

  const goToToday = () => {
    setCurrentWeekStart(getWeekStart(new Date()))
  }

  return {
    currentWeekStart,
    weekTasks,
    tasksByDay,
    goToNextWeek,
    goToPrevWeek,
    goToToday
  }
}

function getWeekStart(date: Date): Date {
  const result = new Date(date)
  const day = result.getDay()
  const diff = day === 0 ? -6 : 1 - day // å‘¨ä¸€ä¸ºä¸€å‘¨å¼€å§‹
  result.setDate(result.getDate() + diff)
  result.setHours(0, 0, 0, 0)
  return result
}
